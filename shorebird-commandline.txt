Android:

⦿ Create Release

APK:
    shorebird release \
    --flutter-version 3.38.7 \
    --platforms=android \
    --artifact=apk  \
    --dart-define-from-file=.env/<flavor>.json \
    --flavor=<flavor> \
    --target=lib/main/main_<flavor>.dart

AAB:
    shorebird release \
    --flutter-version 3.38.7 \
    --platforms=android \
    --artifact=aab  \
    --dart-define-from-file=.env/<flavor>.json \
    --flavor=<flavor> \
    --target=lib/main/main_<flavor>.dart

⦿ Create Preview
    shorebird preview \
    --release-version latest

⦿ Create Patch
    shorebird patch \
    --release-version latest \
    --platforms=android \
    --dart-define-from-file=.env/<flavor>.json \
    --flavor=<flavor> \
    --target=lib/main/main_<flavor>.dart
